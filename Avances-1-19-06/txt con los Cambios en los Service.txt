	public LocalModel getLocal() {
		PedidoModel pedidoAux = new PedidoModel();
		LocalModel local = new LocalModel();
		int ultimoPedido=0;
		List<PedidoModel> pedidos = new ArrayList<PedidoModel>();
		pedidos = getAlls();
		ultimoPedido = pedidos.size();
		pedidoAux = pedidos.get(ultimoPedido-1);
		local = pedidoAux.getLocal();
		return local;
	}

	@Override
	public List<EmpleadoModel> findByIdLocalNoGerente(long idLocal){
		List<EmpleadoModel> empl = new ArrayList<EmpleadoModel>();
		for (Empleado empleado : empleadoRepository.findByIdLocal(idLocal)) {
			if(empleado.isGerente()==false) {
			empl.add(empleadoConverter.entityToModel(empleado));
		}}
		return empl;
	}
